generic

F_h = function(eir, pars){
  UseMethod("F_h", pars$Fh)
}
setup_Fh = function(params, Fname = "pois", phi=5){
  if(Fname == "pois") params = setup_Foi_pois(params) 
  if(Fname == "nb") params = setup_Foi_nb(params, phi) 
}

Poisson

setup_Fh_pois = function(params){
  Fh = list()
  class(Fh) <- 'pois'
  params$Fh = Fh
  return(Fh)
}
F_h.pois = function(eir, pars){
  pars$Xpar$b*eir
}

Negative Binomial

setup_Fh_nb = function(params, phi=10){
  Fh = list()
  class(Fh) <- 'nb'
  Fh$phi <- phi 
  params$Fh = Fh
  return(params)
}
F_h.nb= function(eir, pars){
  # phi=1/size in the mu, size parameterization
  b = pars$Xpar$b
  phi = pars$Fh$phi
  log(1+b*eir*phi)/phi
}
eir = 10^seq(0,3,length.out=100)/365
params = list() 
params$Xpar = list() 
params$Xpar$b = 0.5 
params = setup_Fh_nb(params, 40)
foi = F_h(eir, params)
par(mfrow = c(2,1))
plot(eir*365, foi, log = "x", type = "l", xlab = "aEIR")
plot(eir*365, eir/foi, log = "x", type = "l", xlab = "aEIR")
