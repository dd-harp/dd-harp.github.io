track1 = function(eir, r=1/200, b=0.55, N=25, alpha=40, Tmax=7300){
  tms = 0:Tmax
  prms = c(r=r, b=b, eir=eir, N=N, alpha=alpha) 
  VMC = data.frame(lsode(c(V=0,M=0,C=0), times=tms, trackDE1, prms)) 
  return(VMC)
}
